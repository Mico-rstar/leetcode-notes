# 回溯组合型

### 深化理解
#### 代码模板
给定两个整数 n 和 k，返回范围 [1, n] 中所有可能的 k 个数的组合。
你可以按 任何顺序 返回答案。
1. 选或不选
```Cpp
vecter<vector<int>> combine(int n, int k) {
    vector<vecter<int>> ans;
    vector<int> path;

    auto dfs = [&](this auto&& dfs, int i) -> void {
        int d = k - path.size(); // 还要选d个数
        if(d == 0) {
            ans.emplace_back(path);
            return;
        }

        // 不选
        if(i > d) {
            dfs(i - 1);
        }

        // 选
        path.push_back(i);
        dfs(i - 1);
        path.pop_back();

    };
    dfs(n);
    return ans;
}
```


2. 枚举下一个
```cpp
vecter<vector<int>> combine(int n, int k) {
    vector<vector<int>> ans;
    vector<int> path;

    auto dfs = [&](this auto&& dfs, int i) -> void {
        int d = k - path.size();
        if(d == 0) {
            ans.emplace_back(path);
            return;
        }

        for(int j = i; j >= d; j--) {
            path.push_back(j);
            dfs(j - 1);
            path.pop_back();
        }
    };
    dfs(n);
    return ans;
}
```
